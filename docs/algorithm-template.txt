Input:
    • Sensor streams:

1  function INGEST_AND_ALIGN(S, I_UAV, I_SAT, T, W_f, Σ):
2      D_s   ← read_streams(S)                   // sensor frames with timestamps
3      Iu    ← preprocess_uav(I_UAV)             // radiometric corr, georegister
4      Is    ← preprocess_sat(I_SAT)             // cloud mask, resample
5      Ndvi  ← compute_indices(Iu, Is)           // NDVI/EVI/LAI stacks
6      Traps ← normalize_trap_feed(T)            // counts/species per grid
7      Z     ← zone_map(Σ.fields, Ndvi.bounds)   // vector grid of zones
8      return {D_s, Ndvi, Traps, Z, W_f}

9  function DIAGNOSE_CROP_HEALTH(D_s, Ndvi, Traps, Z, 𝓜):
10     Fz ← fuse_features_by_zone(D_s, Ndvi, Traps, Z)
11     H  ← 𝓜.stress_model.predict(Fz)           // stress class & severity σ
12     C  ← 𝓜.cause_classifier(H, Fz)            // abiotic/biotic/root cause
13     Q  ← quality_metrics(H, C)                // conf, drift, coverage
14     return {H, C, Q, Fz}

15 function PLAN_IRRIGATION(Fz, W_f, Π, 𝓜, Σ):
16     SWD ← compute_swd(Fz.soil, W_f)           // soil water deficit per zone
17     P   ← 𝓜.rl_irrigation.policy(SWD, W_f, Σ.stage)
18     P'  ← apply_rules(P, Π.irrigation)        // rain_prob, wind, tariff windows
19     if confidence(P) < Π.min_conf then P' ← fallback_schedule(Σ)
20     return A_IRR ← finalize_valves(P')

21 function PLAN_PEST_RESPONSE(H, C, Traps, Π, 𝓜, W_f):
22     if C.type = "biotic" or Traps.density > Π.trap_thresh:
23         Rx  ← 𝓜.prescription(H.severity, species=Traps.species, stage=Σ.stage)
24         Rx' ← enforce_label_rules(Rx, Π.labels)   // MRL, PHI, re-entry
25         if W_f.wind > Π.max_wind then return hold_action("wind")
26         Routes ← 𝓜.route_optimizer(Z, no_fly=Π.geofence, buffer=Π.waterways)
27         return A_PEST ← {Rx', Routes}
28     else:
29         return A_PEST ← monitor_only()

30 function FORECAST_YIELD(Fz, Σ, 𝓜):
31     A   ← assemble_timeseries(Fz, ops=Σ.history)
32     Ŷ   ← 𝓜.yield_model.predict(A)              // by zone → lot mapping
33     return A_YLD ← Ŷ

34 function MARKET_DECISIONS(A_YLD, M, Π, Σ):
35     U   ← build_utility(M.prices, costs=M.costs, risk=Π.cvar)
36     Plan← optimize_allocation(A_YLD, capacity=Σ.storage, U, spoilage=Σ.loss)
37     return A_MKT ← Plan

38 function SAFETY_AND_COMPLIANCE(A_IRR, A_PEST, Π):
39     ok1 ← validate_hard_limits(A_IRR, Π.irrigation)
40     ok2 ← validate_safety(A_PEST, Π.labels, Π.weather, Π.geofence)
41     if not(ok1 and ok2) then
42         A_IRR, A_PEST ← degrade_to_safe(A_IRR, A_PEST)
43     return {A_IRR, A_PEST}

44 function EXECUTE_AND_LOG(A_IRR, A_PEST, A_YLD, A_MKT):
45     ids1 ← actuate_valves(A_IRR)                  // edge RPC with acks
46     ids2 ← dispatch_drone(A_PEST)                 // routes + prescriptions
47     ids3 ← schedule_logistics(A_MKT)              // trucks, slots, buyers
48     J    ← justify_with_LLM({A_IRR, A_PEST, A_MKT}, inputs=hash_inputs())
49     Hsh  ← anchor_provenance({ids1, ids2, ids3, J})
50     write_audit({A_IRR, A_PEST, A_YLD, A_MKT, J, Hsh})
51     return {ids1, ids2, ids3, Hsh}

52 // === Main orchestration ===
53 function ANGURAM_ACT_MAIN(S, I_UAV, I_SAT, T, W_f, Π, 𝓜, Σ):
54     D     ← INGEST_AND_ALIGN(S, I_UAV, I_SAT, T, W_f, Σ)
55     H,C,Q,Fz ← DIAGNOSE_CROP_HEALTH(D.D_s, D.Ndvi, D.Traps, D.Z, 𝓜)
56     A_IRR ← PLAN_IRRIGATION(Fz, D.W_f, Π, 𝓜, Σ)
57     A_PEST← PLAN_PEST_RESPONSE(H, C, D.Traps, Π, 𝓜, D.W_f)
58     A_YLD ← FORECAST_YIELD(Fz, Σ, 𝓜)
59     A_MKT ← MARKET_DECISIONS(A_YLD, M=fetch_market(), Π, Σ)
60     A_IRR, A_PEST ← SAFETY_AND_COMPLIANCE(A_IRR, A_PEST, Π)
61     ids  ← EXECUTE_AND_LOG(A_IRR, A_PEST, A_YLD, A_MKT)
62     notify_dashboard({A_IRR, A_PEST, A_YLD, A_MKT, Q}, ids)
63     return {A_IRR, A_PEST, A_YLD, A_MKT}
